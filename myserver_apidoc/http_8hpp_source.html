199 template<class MapType, class T>
200 requires requires (MapType m) {
201 m.find() == m.end();
202 }
203 bool check_get_as(const MapType& m, const std::string& key, T& val, const T& def = T()) {
204 auto it = m.find(key);
205 if(it == m.end()) {
206 val = def;
207 return false;
208 }
209 try {
210 val = boost::lexical_cast<T>(it->second);
211 return true;
212 }
213 catch (...) {
214 val = def;
215 }
216 return false;
217 }

233 return def;
