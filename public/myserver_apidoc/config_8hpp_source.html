130 StlCollectionType operator()(const std::string& src) {

548 static typename ConfigVar<T, FromStr, ToStr>::ptr get_else_insert(std::string name,
549 const T& default_value,
550 const std::string& description = "") {
551 myserver::WriteLock lock(get_mutex());
552 std::transform(name.begin(), name.end(), name.begin(), ::tolower);
553 auto& s_data = get_data();
554 auto it = s_data.find(name);
555 if (it != s_data.end()) {
556 auto tmp = std::dynamic_pointer_cast<ConfigVar<T, FromStr, ToStr>>(it->second);
557 if (tmp) {
558 MYSERVER_LOG_INFO(MYSERVER_ROOT_LOGGER) << "Config variable name = " << name << " exists";
559 return tmp;
560 }
561 else {
562 MYSERVER_LOG_ERROR(MYSERVER_ROOT_LOGGER) << "Config variable name = " << name << "exists, but type is not "
563 << typeid(T).name();
564 return nullptr;
565 }
566 }
567 if (!Config::is_legal_config_name(name)) {
568 MYSERVER_LOG_ERROR(MYSERVER_ROOT_LOGGER) << "get config variable with invalid name: "<< name;
569 throw std::invalid_argument(name);
570 }
571
572 typename ConfigVar<T, FromStr, ToStr>::ptr v(new ConfigVar<T, FromStr, ToStr>(name, default_value, description));
573 s_data[name] = v;
574 return v;
575 }

625 struct LogConfigInitializer {
626 LogConfigInitializer() {
627 Config::get<std::set<LogConfig>>("logs")->add_listener(
628 [](const std::set<myserver::LogConfig>& old_val, const std::set<myserver::LogConfig>& new_val){
629 MYSERVER_LOG_INFO(MYSERVER_ROOT_LOGGER) << "on_logger_conf_changed";
630 for (auto&& i : new_val) {
631 auto it = old_val.find(i);
632 myserver::Logger::ptr logger = nullptr;
633 if (it == old_val.end()) {
634 // 新增 Logger
635 logger = MYSERVER_ADD_LOGGER(i.name);
636 }
637 else if (!(i == *it)) {
638 // 修改 Logger
639 logger = MYSERVER_LOGGER(i.name);
640 }
641 else {
642 // 没有改动
643 continue;
644 }
645 logger->set_level(i.level);
646 if (!i.formatter.empty()) {
647 logger->set_formatter(i.formatter);
648 }
649
650 logger->clear_appenders();
651 for (auto&& a : i.appenders) {
652 myserver::LogAppender::ptr ap = nullptr;
653 if (a.type == 1) {
654 ap.reset(new myserver::FileLogAppender{a.file});
655 }
656 else if (a.type == 2) {
657 ap.reset(new myserver::StdoutLogAppender{});
658 }
659 ap->set_level(a.level);
660 if(!a.formatter.empty()) {
661 myserver::LogFormatter::ptr fmt{new myserver::LogFormatter{a.formatter}};
662 if(!fmt->error()) {
663 ap->set_formatter(fmt);
664 }
665 else {
666 std::cout << "log.name=" << i.name << " appender type=" << a.type
667 << " formatter=" << a.formatter << " is invalid" << std::endl;
668 }
669 }
670 logger->add_appender(ap);
671 }
672 }
673
674 // 删除 Logger
675 for (auto&& i : old_val) {
676 auto it = new_val.find(i);
677 if (it == new_val.end()) {
678 auto logger = MYSERVER_LOGGER(i.name);
679 logger->disable();
680 }
681 }
682 }
683 );
684 } // LogConfigInitializer::LogConfigInitializer()
